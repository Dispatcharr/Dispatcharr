### Feature 2: Profile Failover Fix (3 Bugs)

**Problems:**
1. `current_profile_id` not tracked → same profile retried infinitely
2. `tried_combinations` not used → same stream+profile combo retried
3. `get_alternate_streams()` early break → only tries first profile

**Solution:** Comprehensive failover tracking system

**Files Modified:**

- `apps/proxy/live_proxy/url_utils.py`

`manager.py` _try_next_stream:
```python
def _try_next_stream(self):
    # Prevent infinite loops
    max_attempts = 1000
    if len(self.tried_combinations) >= max_attempts:
        logger.error(f"Safety limit reached: {max_attempts} combinations tried")
        return False
    
    # Get alternate streams
    alternate_streams = url_utils.get_alternate_streams(self.channel)
    
    # Track current combination
    if self.stream_id and self.current_profile_id:
        current_combo = (self.stream_id, self.current_profile_id)
        self.tried_combinations.add(current_combo)
```

    # Redis cooldown system
    if cooldown_enabled:
        redis_keys.stream_cooldown(channel_id, stream_id, profile_id).set(
            cooldown_seconds, nx=True
        )
    
    # Last Resort: Clear all cooldowns after 2 full rounds
    if len(self.tried_combinations) >= 2 * len(alternate_streams):
        logger.warning("Last Resort: Clearing all cooldowns")
        for stream_id, profile_id in self.tried_combinations:
            redis_keys.stream_cooldown(channel_id, stream_id, profile_id).delete()
        self.tried_combinations.clear()
    
    # Try next combination
    for alt_stream, profile in alternate_streams:
        combo = (alt_stream.id, profile.id)
        if combo in self.tried_combinations:
            continue
        
        # Check cooldown
        if cooldown_enabled:
            if redis_keys.stream_cooldown(channel_id, alt_stream.id, profile.id).exists():
                continue
        
        # Switch to this stream+profile
        self.stream_id = alt_stream.id
        self.current_profile_id = profile.id
        self.last_stream_switch_time = time.time()
        return True
    
    return False
```

def get_alternate_streams(channel):
    """Returns list of (stream, profile) tuples for ALL available profiles"""
    result = []
    
    if isinstance(channel, Stream):
        # Single stream case
        profiles = channel.stream_profiles.all()
        for profile in profiles:
            result.append((channel, profile))
    else:
        # Channel with multiple streams
        streams = channel.streams.filter(enabled=True)
        for stream in streams:
            profiles = stream.stream_profiles.all()
            for profile in profiles:
                result.append((stream, profile))
                # REMOVED: break  # This was limiting to first profile!
    
    return result
```

def build_command(self, command_type, input_source, output_file=None, proxy=None):

    command.extend([
        '-timeout', str(timeout * 1000000),  # Convert to microseconds
        '-i', input_source
    ])

    return command

### Feature 6: UUID Validation Fix

```python
def log_system_event(event_type, message, channel=None, stream=None, user=None):
    """
    Log system event with UUID validation fallback
    """
    try:
        # Try to get channel_id as UUID
        if channel:
            try:
                channel_id = uuid.UUID(str(channel.id))
            except (ValueError, AttributeError):
                # Fallback to stream_hash if UUID invalid
                channel_id = getattr(channel, 'stream_hash', None)
        
        # Try to get stream_id as UUID
        if stream:
            try:
                stream_id = uuid.UUID(str(stream.id))
            except (ValueError, AttributeError):
                # Fallback to stream_hash if UUID invalid
                stream_id = getattr(stream, 'stream_hash', None)
        
        # Create log entry
        SystemEvent.objects.create(
            event_type=event_type,
            message=message,
            channel_id=channel_id,
            stream_id=stream_id,
            user=user
        )
    except Exception as e:
        logger.error(f"Failed to log system event: {e}")
```

**Impact:** Prevents crashes when legacy data has non-UUID IDs

### Feature 9: HTTP Reader Race Condition Fix

**Feature:** Synchronize stop_flag and error_occurred between threads

import threading

class HttpStreamer:
    def __init__(self, ...):
        self._stop_flag = threading.Event()  # Thread-safe flag
        self._error_lock = threading.Lock()  # Protect error_occurred
        self.error_occurred = False
    
    def stop(self):
        """Thread-safe stop"""
        self._stop_flag.set()
    
    def _read_stream(self):
        while not self._stop_flag.is_set():
            try:
                # ... read data ...
                pass
            except Exception as e:
                with self._error_lock:
                    self.error_occurred = True
                raise
    
    def has_error(self):
        """Thread-safe error check"""
        with self._error_lock:
            return self.error_occurred
```

**Impact:** Prevents race conditions in multi-threaded streaming

def get_stream_preview_url(channel_or_stream, request):
    """
    Generate preview URL with profile failover
    """
    profiles = []
    
    # Get all profiles for the stream/channel
    if isinstance(channel_or_stream, Stream):
        stream = channel_or_stream
        profiles = list(stream.stream_profiles.all())
    else:
        channel = channel_or_stream
        streams = channel.streams.filter(enabled=True)
        if streams.exists():
            stream = streams.first()
            profiles = list(stream.stream_profiles.all())
    
    # Try each profile until one works
    for profile in profiles:
        try:
            url = profile.get_stream_url()
            
            # Check if URL is accessible
            response = requests.head(
                url,
                timeout=(3, 5),
                allow_redirects=True
            )
            
            if response.status_code < 400:
                return url
                
        except Exception as e:
            logger.debug(f"Profile {profile.id} failed: {e}")
            continue
    
    # Fallback to first profile if all checks failed
    if profiles:
        return profiles[0].get_stream_url()
    
    return None
```

**Impact:** Stream previews work even if default profile is down

`manager.py` _try_next_stream:
```python
def _try_next_stream(self):
    # Get cooldown settings
    cooldown_enabled = config_helper.stream_cooldown_enabled()
    cooldown_seconds = config_helper.stream_cooldown_seconds()
    
    # Mark current combination as failed
    if self.stream_id and self.current_profile_id:
        combo = (self.stream_id, self.current_profile_id)
        self.tried_combinations.add(combo)
        
        # Set cooldown in Redis
        if cooldown_enabled:
            try:
                redis_keys.stream_cooldown(
                    self.channel.id,
                    self.stream_id,
                    self.current_profile_id
                ).set(cooldown_seconds, nx=True)
            except Exception as e:
                logger.warning(f"Redis error (fail-open): {e}")
    
    # Last Resort: Clear all cooldowns after 2 full rounds
    alternate_streams = url_utils.get_alternate_streams(self.channel)
    total_combinations = len(alternate_streams)
    
    if len(self.tried_combinations) >= 2 * total_combinations:
        logger.warning("Last Resort activated: Clearing all cooldowns")
        for stream_id, profile_id in self.tried_combinations:
            try:
                redis_keys.stream_cooldown(
                    self.channel.id,
                    stream_id,
                    profile_id
                ).delete()
            except:
                pass
        self.tried_combinations.clear()
    
    # Find next untried combination
    for stream, profile in alternate_streams:
        combo = (stream.id, profile.id)
        
        # Skip if already tried
        if combo in self.tried_combinations:
            continue
        
        # Skip if in cooldown
        if cooldown_enabled:
            try:
                if redis_keys.stream_cooldown(
                    self.channel.id,
                    stream.id,
                    profile.id
                ).exists():
                    logger.debug(f"Skipping {combo} - in cooldown")
                    continue
            except Exception as e:
                logger.warning(f"Redis error (fail-open): {e}")
                # Continue anyway (fail-open strategy)
        
        # Use this combination
        self.stream_id = stream.id
        self.current_profile_id = profile.id
        self.last_stream_switch_time = time.time()
        return True
    
    # No more combinations available
    return False
```

## 🔒 Security Considerations

### Proxy Credentials
**Risk:** Proxy URLs may contain usernames/passwords  
**Mitigation:**
- Store proxy URLs encrypted in database
- Never log proxy URLs
- Mask proxy credentials in UI

**Example:**
```python
# BAD: Logs full proxy URL with credentials
logger.info(f"Using proxy: {proxy_url}")

masked = re.sub(r'://([^:]+):([^@]+)@', r'://***:***@', proxy_url)

## 🎓 Developer Notes

### Architecture Changes

**Profile Failover:**
- Old: Linear stream iteration (stream1 → stream2 → stream3)
- New: Combinatorial iteration (stream1+profile1 → stream1+profile2 → stream2+profile1...)

**Cooldown System:**
- Storage: Redis (single source of truth)
- Fallback: Fail-open strategy (continue on Redis errors)
- Reset: Last Resort clears all cooldowns after 2 full rounds

**Proxy Architecture:**
- Separation: API proxy vs Streaming proxy
- Injection: Automatic ffmpeg -http_proxy injection
- Timeout: Per-operation timeout configuration
