### Feature 5: build_command() Proxy Fix (CRITICAL)

**Problem:** Transcode streams were completely broken. `manager.py` called `build_command(url, user_agent, proxy)` with 3 arguments, but method accepted only 2.

**Solution in `core/models.py` lines 242-280:**
```python
def build_command(self, stream_url, user_agent, proxy=None):
    replacements = {
        "{streamUrl}": stream_url,
        "{userAgent}": user_agent,
        "{proxy}": proxy or "",  # NEW
    }
    
    cmd = [self.command] + [
        self._replace_in_part(part, replacements)
        for part in shlex_split(self.parameters)
    ]
    
    # Automatic ffmpeg -http_proxy injection
    if proxy and self.command.lower() == 'ffmpeg' and '{proxy}' not in self.parameters:
        try:
            i_index = cmd.index('-i')
            cmd.insert(i_index, proxy)
            cmd.insert(i_index, '-http_proxy')
        except ValueError:
            pass
    
    return cmd
```

### Feature 6: UUID Validation Fix

**Problem:** Stream Preview channels use `stream_hash` (not UUID format), causing database errors.

**Solution in `core/utils.py` lines 185-200:**
```python
def log_system_event(event_type, channel_id=None, channel_name=None, **details):
    safe_channel_id = None
    if channel_id is not None:
        try:
            uuid_module.UUID(str(channel_id))
            safe_channel_id = channel_id
        except (ValueError, AttributeError):
            # Not a valid UUID (e.g., stream_hash for stream preview)
            details['stream_hash'] = str(channel_id)
            logger.debug(f"channel_id '{channel_id}' is not a valid UUID")
    
    SystemEvent.objects.create(
        event_type=event_type,
        channel_id=safe_channel_id,  # Use safe_channel_id instead
        channel_name=channel_name,
        details=details
    )
```

```python
def get_basic_auth_user(request):
    """Extract and validate user from Basic Auth header"""
    auth_header = request.META.get('HTTP_AUTHORIZATION', '')
    if not auth_header.startswith('Basic '):
        return None
    
    try:
        decoded = base64.b64decode(auth_header[6:]).decode('utf-8')
        username, password = decoded.split(':', 1)
        user = authenticate(username=username, password=password)
        return user
    except Exception:
        return None

def require_basic_auth(request):
    """Return 401 with WWW-Authenticate if no valid auth"""
    response = HttpResponse('Unauthorized', status=401)
    response['WWW-Authenticate'] = 'Basic realm="Dispatcharr"'
    return response
```

### Feature 15: Buffer Timeout Failover

**File:** `apps/proxy/live_proxy/server.py` lines 1808-1836

**Status:** ✅ ALREADY IMPLEMENTED IN v0.27.0!

**Code:**
```python
if time_since_start > connecting_timeout:
    # BUGFIX: Trigger failover instead of stopping immediately
    stream_manager = self.stream_managers.get(channel_id)
    if stream_manager and not getattr(stream_manager, 'url_switching', False):
        logger.warning(
            f"Channel {channel_id} stuck in {channel_state} state for {time_since_start:.1f}s "
            f"(timeout: {connecting_timeout}s) - triggering failover"
        )
        try:
            switch_success = stream_manager._try_next_stream()
            if switch_success:
                logger.info(f"Buffer timeout failover triggered successfully")
            else:
                logger.warning(f"Buffer timeout failover failed - no alternate streams")
                self._coordinated_stop_channel(channel_id)
        except Exception as e:
            logger.error(f"Error during buffer timeout failover: {e}")
            self._coordinated_stop_channel(channel_id)
        continue
```
